! Set IP address 

sudo vi /etc/network/interfaces

setup-hostname -n LINUX14
auto lo
iface lo inet loopback

auto eth0
iface eth0 inet static
    address 192.168.X.X
    netmask 255.255.255.0
    gateway 192.168.X.1


# ifdown eth0
# ifup eth0


ip addr flush dev eth0
ip addr add 192.168.X.X/24 dev eth0
ip route add default via 192.168.X.1





# this script is sourced at boot by CML

# hostname (optional)
hostname alpine1
echo alpine1 > /etc/hostname

# configure static IP on eth0
echo "auto lo
iface lo inet loopback

auto eth0
iface eth0 inet static
    address 192.168.X.X
    netmask 255.255.255.0
    gateway 192.168.X.1" > /etc/network/interfaces

# enable networking at boot (required in CML Alpine)
rc-update add networking boot

# DNS (optional)
echo 'nameserver 8.8.8.8' > /etc/resolv.conf
